# アルゴ式用の入力(個人用)

```hs
-- 入力1つ(IntはBsやDoubleでもOK)
a <- input @Int

-- 入力1行複数
a <- input @[Int]
-- 1 2 3 4 -> [1,2,3,4] 

-- 複数行複数文字
a <- inputs @[[Int]]
{-
          [
1 2 3      [1,2,3],
4 5 6  =>  [4,5,6],
7 8 9      [7,8,9]
          ]
-}  

-- 複数行２列 (前と後ろが数値もしくは文字列)の時
ab <- inputs @[(Int,Bs)]

{-
                        [
Everest 8849              ("Everest",8849),
K2 8611              =>   ("K2",8611),
Kangchenjunga 8586        ("Kangchenjunga",8586)
                        ]
-}

-- 2次元Arrayで受け取る(これいるか？)
a <- inputArray @(Array2D Int) 3 3
{-
           
2 3            array((0,0),(1,2))[
1 2 3          ((0,0),1),((0,1),2),((0,2),3),
2 4 6   =>   ((1,0),2),((1,1),4),((1,2),6)
               ]           
-}
```
